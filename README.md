# VueJS VKUI

## Installation
```
npm install @urapywka/vkui
```

### Usage
```
import Vue from 'vue'
...

import '@urapywka/vkui'

import '@urapywka/vkui-icons'
import '@urapywka/vkui-icons/dist/vkui-icons.css'

import connect from '@vkontakte/vkui-connect'

import PortalVue from 'portal-vue'
Vue.use(PortalVue)
...

```
